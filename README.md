Database Schema (Supabase)Run this SQL in your Supabase Editor to create the table:create table practice_notes (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  hole_number bigint,
  layout text,
  disc_used text,
  result_rating bigint,
  notes text
);
alter table practice_notes enable row level security;
create policy "Allow public access" on practice_notes for all using (true) with check (true);
